tabla <- read_excel("SISVER 28042020.xlsx")
tabla<-tabla %>% rename(Municipio=MPIORESI,Sexo=SEXO,Institucion=SECTOR) %>%
  mutate(Evolucion=factor(case_when(
    EVOLUCI=="CASO GRAVE -"  ~ "Hospitalizados graves",
    EVOLUCI=="DEFUNCION"  ~ "Defunciones",
    EVOLUCI=="SEGUIMIENTO TERMINADO" |  EVOLUCI=="ALTA - MEJORIA" ~ "Recuperados",
    EVOLUCI=="EN TRATAMIENTO" |  EVOLUCI=="SEGUIMIENTO DOMICILIARIO" | EVOLUCI=="CASO NO GRAVE"~ "Ambulatorios y leves",
      TRUE                      ~ "Otros"
  )),Diagnostico=factor(case_when(
    RESDEFIN=="SARS-CoV-2"  ~ "Positivos",
    RESDEFIN=="RECHAZADA" | RESDEFIN=="NO RECIBIDA" ~ "Sospechosos",
    TRUE ~ "Negativos"
  )),Edad=case_when(
    EDAD<=4 ~"1 a 4 a√±os",
    EDAD>=5 | EDAD<=14  ~"5 a 14 a√±os",
    EDAD>=15 | EDAD<=19  ~"5 a 19 a√±os",
    EDAD>=20 | EDAD<=24  ~"20 a 24 a√±os",
    EDAD>=25 | EDAD<=44  ~"25 a 44 a√±os",
    EDAD>=45 | EDAD<=49  ~"45 a 49 a√±os",
    EDAD>=50 | EDAD<=59  ~"50 a 59 a√±os",
    EDAD>=60 | EDAD<=64  ~"60 a 64 a√±os",
    TRUE ~ "65+"
    
  ),Sexo=factor(case_when(
    Sexo=="MASCULINO" ~"Masculino",
    TRUE ~ "Femenino"
  )))
